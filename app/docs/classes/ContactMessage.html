export class ContactMessage {
  /**
   * Sender name
   */
  public name: string;
  /**

   */
  public email: string;
  /**
   * Message content
   */
  public text: string;
  /**
   * Anti-spam field
   */
  public heuning: string;

  constructor(name?: string, email?: string, text?: string, heuning?: string) {
    this.name = name;
    this.email = email;
    this.text = text;
    this.heuning = heuning;
  }
}
